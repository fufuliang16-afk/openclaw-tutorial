OpenClaw技能开发指南

学习如何创建自定义技能，扩展OpenClaw的能力。

什么是技能？

技能定义

技能是OpenClaw的可扩展模块，让AI能够执行特定任务：

  • 工具封装 - 将复杂操作封装为简单指令
  • 工作流自动化 - 创建自动化任务流程
  • 外部集成 - 连接第三方API和服务
  • 专业知识 - 提供特定领域的知识
技能类型
  • 文件操作技能 - 处理文档、图片、数据
  • 网络技能 - 搜索、爬虫、API调用
  • 系统技能 - 命令执行、进程管理
  • 通信技能 - 邮件、消息、通知
  • 专业领域技能 - 编程、设计、分析

## 描述
统计目录中的文件数量和类型的技能。

## 触发关键词
- 文件统计
- 文件数量
- 目录分析
- 文件类型

## 使用示例
```bash
openclaw "统计当前目录的文件"

```

## 功能
- 统计文件总数
- 按类型分类文件
- 计算文件大小
- 生成统计报告

// src/file-stats.js
const fs = require('fs');
const path = require('path');

class FileStatsSkill {
  constructor() {
    this.name = '文件统计技能';
    this.description = '统计目录中的文件信息';
  }

  async execute(directory = '.') {
    const stats = {
      totalFiles: 0,
      totalSize: 0,
      byType: {},
      byExtension: {}
    };

    const files = fs.readdirSync(directory);
    
    for (const file of files) {
      const filePath = path.join(directory, file);
      const stat = fs.statSync(filePath);
      
      if (stat.isFile()) {
        stats.totalFiles++;
        stats.totalSize += stat.size;
        
        const ext = path.extname(file).toLowerCase() || '无扩展名';
        stats.byExtension[ext] = (stats.byExtension[ext] || 0) + 1;
        
        // 按类型分类
        const fileType = this.getFileType(ext);
        stats.byType[fileType] = (stats.byType[fileType] || 0) + 1;
      }
    }
    
    return this.formatReport(stats);
  }

  getFileType(extension) {
    const typeMap = {
      '.js': 'JavaScript',
      '.py': 'Python',
      '.java': 'Java',
      '.cpp': 'C++',
      '.md': 'Markdown',
      '.json': 'JSON',
      '.txt': '文本',
      '.pdf': 'PDF',
      '.jpg': '图片',
      '.png': '图片',
      '.gif': '图片',
      '.mp4': '视频',
      '.mp3': '音频'
    };
    
    return typeMap[extension] || '其他';
  }

  formatReport(stats) {
    let report = `📊 文件统计报告\n`;
    report += `📁 文件总数: ${stats.totalFiles}\n`;
    report += `💾 总大小: ${(stats.totalSize / 1024 / 1024).toFixed(2)} MB\n\n`;
    
    report += `📂 按类型分类:\n`;
    for (const [type, count] of Object.entries(stats.byType)) {
      report += `  ${type}: ${count} 个文件\n`;
    }
    
    report += `\n🔤 按扩展名分类:\n`;
    for (const [ext, count] of Object.entries(stats.byExtension)) {
      report += `  ${ext}: ${count} 个文件\n`;
    }
    
    return report;
  }
}

module.exports = FileStatsSkill;

// src/index.js
const FileStatsSkill = require('./file-stats');

module.exports = {
  name: 'file-stats',
  description: '文件统计技能',
  
  // 技能初始化
  async init() {
    this.skill = new FileStatsSkill();
    console.log('✅ 文件统计技能已加载');
  },
  
  // 技能执行
  async execute(params) {
    const directory = params.directory || '.';
    return await this.skill.execute(directory);
  },
  
  // 技能匹配
  matches(input) {
    const keywords = ['统计文件', '文件数量', '目录分析', '文件统计'];
    return keywords.some(keyword => input.includes(keyword));
  }
};

class WeatherSkill {
  async getWeather(city) {

      `https://api.weatherapi.com/v1/current.json?key=YOUR_KEY&q=${city}`
    );
    const data = await response.json();
    
    return `🌤️ ${city}天气：
    温度: ${data.current.temp_c}°C
    天气: ${data.current.condition.text}
    湿度: ${data.current.humidity}%
    风速: ${data.current.wind_kph} km/h`;
  }
}

// 简化的Git技能
const { execSync } = require('child_process');

class GitSkill {
  getStatus() {
    return execSync('git status', { encoding: 'utf8' });
  }
  
  commit(message) {
    execSync(`git add . && git commit -m "${message}"`);
    return '✅ 提交完成';
  }
  
  push(branch = 'main') {
    execSync(`git push origin ${branch}`);
    return `✅ 已推送到 ${branch} 分支`;
  }
}
